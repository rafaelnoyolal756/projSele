
--poner el esquema asociado para cada objeto
CREATE TABLE CATALOG(
ID PRIMARY KEY NOT NULL,
NAME VARCHAR(200),
DATA BLOB(),
CURRENT_TYPE VARCHAR(100),
HISTORICAL_TYPE VARCHAR(100),
START_DATE DATE,
END_DATE DATE,
VERSION VARCHAR(100)
);

ON CATALOG (ID);
